592 static inline int
593 inline_mysql_socket_bind
594 (
595 #ifdef HAVE_PSI_SOCKET_INTERFACE
596  const char *src_file, uint src_line,
597 #endif
598  MYSQL_SOCKET mysql_socket, const struct sockaddr *addr, socklen_t len)
599 {
600  int result;
601 
602 #ifdef HAVE_PSI_SOCKET_INTERFACE
603  if (mysql_socket.m_psi != NULL)
604  {
605  /* Instrumentation start */
606  PSI_socket_locker_state state;
607  PSI_socket_locker *locker;
608  locker= PSI_SOCKET_CALL(start_socket_wait)
609  (&state, mysql_socket.m_psi, PSI_SOCKET_BIND, (size_t)0, src_file, src_line);
610 
611  /* Instrumented code */
612  result= bind(mysql_socket.fd, addr, len);
613 
614  /* Instrumentation end */
615  if (result == 0)
616  PSI_SOCKET_CALL(set_socket_info)(mysql_socket.m_psi, NULL, addr, len);
617 
618  if (locker != NULL)
619  PSI_SOCKET_CALL(end_socket_wait)(locker, (size_t)0);
620 
621  return result;
622  }
623 #endif
624 
625  /* Non instrumented code */
626  result= bind(mysql_socket.fd, addr, len);
627  return result;
628 }
